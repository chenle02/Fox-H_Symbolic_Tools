\paragraph{Fox H-function}

\begin{align*}
  \FoxH
    {2,1}
    {1,2}
    {\cdot}
    {\left(\gamma -\frac{1}{c}+1, \frac{1}{c}\right)}
    {\left(c \gamma, 1\right), \left(0, \frac{1}{c}\right)}
\end{align*}

\begin{align*}
  \FoxHext
    {2,1}

    {\cdot}
    {\left(\gamma -\frac{1}{c}+1, \frac{1}{c}\right)}
    {}
    {\left(c \gamma, 1\right), \left(0, \frac{1}{c}\right)}
    {}
\end{align*}

\paragraph{Summary}

\begin{align*}
  a^*    & = \frac{c+2}{c} \\
  \Delta & = 1 \\
  \delta & = \text{Indeterminate} \\
  \mu    & = c \gamma -\gamma +\frac{1}{c}-\frac{3}{2} \\
  a_1^*  & = \frac{1}{c}+1 \\
  a_2^*  & = \frac{1}{c} \\
  \xi    & = c \gamma +\gamma -\frac{1}{c}+1 \\
  c^*    & = \frac{3}{2} \\
\end{align*}
