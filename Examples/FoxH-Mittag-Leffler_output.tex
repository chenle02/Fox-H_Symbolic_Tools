\paragraph{Fox H-function}

\begin{align*}
  \FoxH
    {1,1}
    {1,2}
    {\cdot}
    {\left(0, 1\right)}
    {\left(0, 1\right), \left(1-\mu, \rho\right)}
\end{align*}

\begin{align*}
  \FoxHext
    {1,1}
    {1,2}
    {\cdot}
    {\left(0, 1\right)}
    {}
    {\left(0, 1\right)}
    {\left(1-\mu, \rho\right)}
\end{align*}

\paragraph{Summary}

\begin{align*}
  a^*    & = 2-\rho \\
  \Delta & = \rho \\
  \delta & = \rho ^{\rho } \\
  \mu    & = \frac{1}{2}-\mu \\
  a_1^*  & = 1 \\
  a_2^*  & = 1-\rho \\
  \xi    & = \mu -1 \\
  c^*    & = \frac{1}{2} \\
\end{align*}
