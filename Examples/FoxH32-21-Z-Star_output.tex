\paragraph{Summary}

\begin{align*}
  a^*    & = 2-\beta \\
  \Delta & = \alpha -\beta \\
  \delta & = 2^{-\alpha } \left(2^{\alpha /2} \alpha ^{\alpha /2}+\alpha ^{\alpha }\right) \beta ^{-\beta } \\
  \mu    & = \frac{d-1}{2} \\
  a_1^*  & = \frac{1}{2} (\alpha -2 \beta +2) \\
  a_2^*  & = 1-\frac{\alpha }{2} \\
  \xi    & = \frac{d}{2} \\
  c^*    & = \frac{1}{2} \\
\end{align*}
